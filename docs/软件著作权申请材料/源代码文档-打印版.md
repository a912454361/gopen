import React, { createContext, useContext, useState, useEffect } from 'react';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { supabase } from '../storage/database/supabase-client';

interface User {
  id: string;
  phone: string;
  nickname?: string;
  avatar_url?: string;
  membership_type: 'free' | 'normal' | 'super';
  membership_expire_at?: string;
  created_at: string;
}

interface AuthContextType {
  user: User | null;
  isLoading: boolean;
  isAuthenticated: boolean;
  login: (phone: string, code: string) => Promise<void>;
  logout: () => Promise<void>;
  refreshUser: () => Promise<void>;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be used within an AuthProvider');
  }
  return context;
};

export const AuthProvider: React.FC<{ children: ReactNode }> = ({ children }) => {
  const [user, setUser] = useState<User | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    checkAuth();
  }, []);

  const checkAuth = async () => {
    try {
      const userId = await AsyncStorage.getItem('userId');
      if (userId) {
        const { data, error } = await supabase
          .from('users')
          .select('*')
          .eq('id', userId)
          .single();
        if (!error && data) {
          setUser(data);
        }
      }
    } catch (error) {
      console.error('Check auth error:', error);
    } finally {
      setIsLoading(false);
    }
  };

  const login = async (phone: string, code: string) => {
    const response = await fetch('/api/v1/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ phone, code }),
    });
    if (!response.ok) {
      throw new Error('登录失败');
    }
    const { user: userData, token } = await response.json();
    await AsyncStorage.setItem('userId', userData.id);
    await AsyncStorage.setItem('token', token);
    setUser(userData);
  };

  const logout = async () => {
    await AsyncStorage.removeItem('userId');
    await AsyncStorage.removeItem('token');
    setUser(null);
  };

  return (
    <AuthContext.Provider value={{ user, isLoading, isAuthenticated: !!user, login, logout, refreshUser }}>
      {children}
    </AuthContext.Provider>
  );
};

      'priority_support': ['super'],
    };
    return featureLevels[feature]?.includes(level) ?? false;
  };

  const refresh = async () => {
    // 刷新会员状态
  };

  return (
    <MembershipContext.Provider value={{ level, expireAt, isActive, canUseFeature, refresh }}>
      {children}
    </MembershipContext.Provider>
  );

/**
 * 主题Hook
 * 提供主题切换功能
 */

import { useState, useEffect, useMemo } from 'react';
import { Appearance } from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { Theme, darkTheme, lightTheme } from '@/constants/theme';

const THEME_KEY = 'app_theme';
type ThemeMode = 'light' | 'dark' | 'system';

export const useTheme = () => {
  const [mode, setMode] = useState<ThemeMode>('system');
  const [systemColorScheme, setSystemColorScheme] = useState<'light' | 'dark'>('light');

  useEffect(() => {
    loadTheme();
    const subscription = Appearance.addChangeListener(({ colorScheme }) => {
      setSystemColorScheme(colorScheme ?? 'light');
    });
    return () => subscription.remove();
  }, []);

  const loadTheme = async () => {
    const savedMode = await AsyncStorage.getItem(THEME_KEY) as ThemeMode | null;
    if (savedMode) {
      setMode(savedMode);
    }
  };

  const setThemeMode = async (newMode: ThemeMode) => {
    setMode(newMode);
    await AsyncStorage.setItem(THEME_KEY, newMode);
  };

  const isDark = useMemo(() => {
    if (mode === 'system') {
      return systemColorScheme === 'dark';
    }
    return mode === 'dark';
  }, [mode, systemColorScheme]);

  const theme = useMemo(() => {
    return isDark ? darkTheme : lightTheme;
  }, [isDark]);

  return { theme, isDark, mode, setThemeMode };
};

/**
 * 安全路由Hook
 * 处理参数编码解码
 */

import { useRouter, useLocalSearchParams } from 'expo-router';
import { useCallback, useMemo } from 'react';

interface SafeRouter {
  push: (href: string, params?: Record<string, unknown>) => void;
  replace: (href: string, params?: Record<string, unknown>) => void;
  back: () => void;
  navigate: (href: string) => void;
}

export const useSafeRouter = (): SafeRouter => {
  const router = useRouter();

  const encodeParams = (params?: Record<string, unknown>): string => {
    if (!params) return '';
    const encoded = Object.entries(params)
      .map(([key, value]) => {
        const encodedValue = encodeURIComponent(
          typeof value === 'object' ? JSON.stringify(value) : String(value)
        );
        return `${key}=${encodedValue}`;
      })
      .join('&');
    return encoded ? `?${encoded}` : '';
  };

  const push = useCallback((href: string, params?: Record<string, unknown>) => {
    const url = `${href}${encodeParams(params)}`;
    router.push(url as any);
  }, [router]);

  const replace = useCallback((href: string, params?: Record<string, unknown>) => {
    const url = `${href}${encodeParams(params)}`;
    router.replace(url as any);
  }, [router]);

  const back = useCallback(() => {
    router.back();
  }, [router]);

  const navigate = useCallback((href: string) => {
    router.navigate(href as any);
  }, [router]);

  return { push, replace, back, navigate };
};

export const useSafeSearchParams = <T extends Record<string, unknown>>(): T => {
  const params = useLocalSearchParams();
  return useMemo(() => {
    const result: Record<string, unknown> = {};
    for (const [key, value] of Object.entries(params)) {
      if (typeof value === 'string') {
        try {
          result[key] = JSON.parse(decodeURIComponent(value));
        } catch {
          result[key] = decodeURIComponent(value);
        }
      } else {
        result[key] = value;
      }
    }
    return result as T;
  }, [params]);
};

import React, { ReactNode } from 'react';
import { View, ViewProps } from 'react-native';
import { useTheme } from '@/hooks/useTheme';

type ViewLevel = 'root' | 'default' | 'tertiary';

interface ThemedViewProps extends ViewProps {
  children: ReactNode;
  level?: ViewLevel;
}

export const ThemedView: React.FC<ThemedViewProps> = ({
  children,
  level = 'default',
  style,
  ...props
}) => {
  const { theme } = useTheme();

  const getBackgroundColor = () => {
    switch (level) {
      case 'root':
        return theme.backgroundRoot;
      case 'default':
        return theme.backgroundDefault;
      case 'tertiary':
        return theme.backgroundTertiary;
      default:
        return theme.backgroundDefault;
    }
  };

  return (
    <View
      style={[
        { backgroundColor: getBackgroundColor() },
        style,
      ]}
      {...props}
    >
      {children}
    </View>
  );
};

  timestamp: Date;

export default function ChatScreen() {
  const { theme, isDark } = useTheme();
  const { user } = useAuth();
  const { canUseFeature } = useMembership();
  const [messages, setMessages] = useState<Message[]>([]);
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const scrollViewRef = useRef<ScrollView>(null);

  const sendMessage = async () => {
    if (!input.trim() || isLoading) return;
    
    const userMessage: Message = {
      id: Date.now().toString(),
      role: 'user',
      content: input.trim(),
      timestamp: new Date(),
    };
    
    setMessages(prev => [...prev, userMessage]);
    setInput('');
    setIsLoading(true);

    try {
      const response = await fetch('/api/v1/ai/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          message: userMessage.content,
          history: messages.slice(-10),
        }),
      });

      const data = await response.json();
      
      const assistantMessage: Message = {
        id: (Date.now() + 1).toString(),
        role: 'assistant',
        content: data.reply,
        timestamp: new Date(),
      };
      
      setMessages(prev => [...prev, assistantMessage]);
    } catch (error) {
      console.error('Chat error:', error);
    } finally {
      setIsLoading(false);
    }
  };

  useEffect(() => {
    scrollViewRef.current?.scrollToEnd({ animated: true });

const IS_DESKTOP = width >= 1024;

type AdminTab = 'stats' | 'users' | 'orders' | 'profit' | 'promotion' | 'logs' | 'config';

export default function AdminScreen() {
  const { theme, isDark } = useTheme();
  const router = useSafeRouter();
  const [activeTab, setActiveTab] = useState<AdminTab>('stats');
  const [isAuthenticated, setIsAuthenticated] = useState(false);

  useEffect(() => {

    const adminToken = await AsyncStorage.getItem('adminToken');

    return null;
