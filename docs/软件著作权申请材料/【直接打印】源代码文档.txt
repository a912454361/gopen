export const AuthProvider: React.FC<AuthProviderProps> = ({ children }) => {
  const [user, setUser] = useState<User | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    checkAuth();
  }, []);

  const checkAuth = async () => {
    try {
      const userId = await AsyncStorage.getItem('userId');
      if (userId) {
        const { data, error } = await supabase
          .from('users')
          .select('*')
          .eq('id', userId)
          .single();
        
        if (!error && data) {
          setUser(data);
        }
      }
    } catch (error) {
      console.error('Check auth error:', error);
    } finally {
      setIsLoading(false);
    }
  };

import { useState, useEffect, useMemo } from 'react';
import { Appearance } from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { Theme, darkTheme, lightTheme } from '@/constants/theme';

const THEME_KEY = 'app_theme';
type ThemeMode = 'light' | 'dark' | 'system';

export const useTheme = () => {
  const [mode, setMode] = useState<ThemeMode>('system');
  const [systemColorScheme, setSystemColorScheme] = useState<'light' | 'dark'>('light');

  useEffect(() => {
    loadTheme();
    const subscription = Appearance.addChangeListener(({ colorScheme }) => {
      setSystemColorScheme(colorScheme ?? 'light');
    });
    return () => subscription.remove();
  }, []);

  const loadTheme = async () => {
    const savedMode = await AsyncStorage.getItem(THEME_KEY) as ThemeMode | null;
    if (savedMode) {
      setMode(savedMode);
    }
  };

  const setThemeMode = async (newMode: ThemeMode) => {
    setMode(newMode);
    await AsyncStorage.setItem(THEME_KEY, newMode);
  };

  const isDark = useMemo(() => {
    if (mode === 'system') {
      return systemColorScheme === 'dark';
    }
    return mode === 'dark';
  }, [mode, systemColorScheme]);

  const theme = useMemo(() => {
    return isDark ? darkTheme : lightTheme;
  }, [isDark]);

  return {
    theme,
    isDark,
    mode,
    setThemeMode,
  };
};

interface SafeRouter {
  push: (href: string, params?: Record<string, unknown>) => void;
  replace: (href: string, params?: Record<string, unknown>) => void;
  back: () => void;
  navigate: (href: string) => void;
}

export const useSafeRouter = (): SafeRouter => {
  const router = useRouter();

  const encodeParams = (params?: Record<string, unknown>): string => {
    if (!params) return '';
    const encoded = Object.entries(params)
      .map(([key, value]) => {
        const encodedValue = encodeURIComponent(
          typeof value === 'object' ? JSON.stringify(value) : String(value)
        );
        return `${key}=${encodedValue}`;
      })
      .join('&');
    return encoded ? `?${encoded}` : '';
  };

  const push = useCallback((href: string, params?: Record<string, unknown>) => {
    const url = `${href}${encodeParams(params)}`;
    router.push(url as any);
  }, [router]);

  const replace = useCallback((href: string, params?: Record<string, unknown>) => {
    const url = `${href}${encodeParams(params)}`;
    router.replace(url as any);
  }, [router]);

  const back = useCallback(() => {
    router.back();
  }, [router]);

  const navigate = useCallback((href: string) => {
    router.navigate(href as any);
  }, [router]);

  return { push, replace, back, navigate };
};

export const useSafeSearchParams = <T extends Record<string, unknown>>(): T => {
  const params = useLocalSearchParams();
  
  return useMemo(() => {
    const result: Record<string, unknown> = {};
    for (const [key, value] of Object.entries(params)) {
      if (typeof value === 'string') {
        try {
          result[key] = JSON.parse(decodeURIComponent(value));
        } catch {
          result[key] = decodeURIComponent(value);
        }
      } else {
        result[key] = value;
      }
    }
    return result as T;
  }, [params]);
};

import axios from 'axios';

const SMS_API_URL = process.env.SMS_API_URL || '';
const SMS_API_KEY = process.env.SMS_API_KEY || '';

const codeStore = new Map<string, { code: string; expireAt: number }>();

export const sendSmsCode = async (phone: string): Promise<void> => {
  const code = Math.random().toString().slice(-6);
  const expireAt = Date.now() + 5 * 60 * 1000;

  codeStore.set(phone, { code, expireAt });

  if (SMS_API_URL && SMS_API_KEY) {
    await axios.post(SMS_API_URL, {
      phone,
      template: 'verification_code',
      params: { code },
    }, {
      headers: { Authorization: `Bearer ${SMS_API_KEY}` },
    });
  }

  console.log(`[SMS] Code for ${phone}: ${code}`);
};

export const verifySmsCode = async (phone: string, code: string): Promise<boolean> => {
  const stored = codeStore.get(phone);
  if (!stored) return false;

  if (Date.now() > stored.expireAt) {
    codeStore.delete(phone);
    return false;
  }

  if (stored.code !== code) return false;

  codeStore.delete(phone);
  return true;
};

import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

const JWT_SECRET = process.env.JWT_SECRET || 'gopen_secret_key_2024';

export interface AuthRequest extends Request {
  userId?: string;
}

export const authMiddleware = (
  req: AuthRequest,
  res: Response,
  next: NextFunction
): void => {
  const token = req.headers.authorization?.replace('Bearer ', '');

  if (!token) {
    res.status(401).json({ error: '请先登录' });
    return;
  }

  try {
    const decoded = jwt.verify(token, JWT_SECRET) as { userId: string };
    req.userId = decoded.userId;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Token无效' });
  }
};

export const adminMiddleware = (
  req: Request,
  res: Response,
  next: NextFunction
): void => {

    res.status(403).json({ error: '无权限' });
    return;
  }

  next();
};

import fs from 'fs';
import path from 'path';

const LOG_DIR = process.env.LOG_DIR || '/app/work/logs';

type LogLevel = 'info' | 'warn' | 'error' | 'debug';

interface LogEntry {
  timestamp: string;
  level: LogLevel;
  message: string;
  data?: any;
}

const formatLog = (entry: LogEntry): string => {
  const dataStr = entry.data ? ` ${JSON.stringify(entry.data)}` : '';
  return `[${entry.timestamp}] [${entry.level.toUpperCase()}] ${entry.message}${dataStr}\n`;
};

const writeLog = (level: LogLevel, message: string, data?: any): void => {
  const entry: LogEntry = {
    timestamp: new Date().toISOString(),
    level,
    message,
    data,
  };

  const logPath = path.join(LOG_DIR, 'app.log');
  const logLine = formatLog(entry);

  fs.appendFileSync(logPath, logLine);

  if (level === 'error') {
    console.error(logLine.trim());
  } else if (level === 'warn') {
    console.warn(logLine.trim());
  } else {
    console.log(logLine.trim());
  }
};

export const logger = {
  info: (message: string, data?: any) => writeLog('info', message, data),
  warn: (message: string, data?: any) => writeLog('warn', message, data),
  error: (message: string, data?: any) => writeLog('error', message, data),
  debug: (message: string, data?: any) => writeLog('debug', message, data),
};
